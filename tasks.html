مشروع لعبة اسءلة

مشروع لعية الاسءلة نصي بي تكرار حاول تعمله بنفسك

كود مشروع آلة حاسبة نصية

                            # Questions and correct answers
                            questions = {
                                "What is the largest planet in the solar system?": "Jupiter",
                                "What is the smallest animal in the world?": "Bumblebee bat",
                                "Who invented the telephone?": "Alexander Graham Bell",
                                "What is the capital of France?": "Paris"
                            }
                            
                            # While loop to repeat the questions
                            while True:
                                # Pick a question randomly from the dictionary
                                for question, correct_answer in questions.items():
                                    print(question)
                                    user_answer = input("Enter your answer: ").strip()
                                    
                                    # Check if the answer is correct
                                    if user_answer.lower() == correct_answer.lower():
                                        print("Correct answer!")
                                    else:
                                        print(f"Incorrect answer! The correct answer is: {correct_answer}")
                                    
                                    # Ask the user if they want to continue
                                    continue_game = input("Do you want to continue playing? (yes/no): ").strip().lower()
                                    
                                    if continue_game != "yes":
                                        print("Thanks for playing!")
                                        break
                            
                                # Ask the user if they want to restart the game
                                restart_game = input("Do you want to restart the game? (yes/no): ").strip().lower()
                                if restart_game != "yes":
                                    print("Goodbye!")
                                    break

مشروع 3: برنامج إدارة المهام

وصف: برنامج يسمح للمستخدم بإضافة مهام وتنظيمها.

كود مشروع إدارة المهام

# برنامج إدارة المهام
tasks = []

def add_task(task):
    tasks.append(task)
    print("تم إضافة المهمة: ", task)

def show_tasks():
    print("المهام الحالية:")
    for task in tasks:
        print("- ", task)

# إضافة مهام وعرضها
add_task("مراجعة المشروع")
add_task("قراءة كتاب")
show_tasks()
