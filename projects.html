آلة حاسبة بسيطة

في هذا المشروع، سنقوم بإنشاء آلة حاسبة بسيطة باستخدام بايثون. يمكن للمستخدم اختيار العملية الحسابية (جمع، طرح، ضرب، قسمة) وإدخال الأرقام التي يرغب في العمل عليها. هذا المشروع هو نقطة انطلاق جيدة لفهم كيفية استخدام بايثون لتنفيذ العمليات الرياضية الأساسية.


def add(x, y):
    return x + y

def subtract(x, y):
    return x - y

def multiply(x, y):
    return x * y

def divide(x, y):
    if y == 0:
        return "لا يمكن القسمة على صفر"
    return x / y

print("اختر العملية:")
print("1. جمع")
print("2. طرح")
print("3. ضرب")
print("4. قسمة")

choice = input("أدخل رقم الخيار (1/2/3/4): ")

num1 = float(input("أدخل الرقم الأول: "))
num2 = float(input("أدخل الرقم الثاني: "))

if choice == '1':
    print(f"النتيجة: {add(num1, num2)}")
elif choice == '2':
    print(f"النتيجة: {subtract(num1, num2)}")
elif choice == '3':
    print(f"النتيجة: {multiply(num1, num2)}")
elif choice == '4':
    print(f"النتيجة: {divide(num1, num2)}")
else:
    print("خيار غير صالح")

مولد كلمات مرور عشوائية

هذا المشروع يقوم بإنشاء كلمة مرور عشوائية تحتوي على أحرف كبيرة وصغيرة، أرقام ورموز خاصة. الهدف من هذا المشروع هو تعلم كيفية التعامل مع النصوص والأحرف العشوائية في بايثون، وكذلك كيفية استخدام مكتبة `string` لإنشاء كلمات مرور آمنة.


import random
import string

def generate_password(length):
    characters = string.ascii_letters + string.digits + string.punctuation
    password = ''.join(random.choice(characters) for i in range(length))
    return password

length = int(input("أدخل طول كلمة المرور: "))
print(f"كلمة المرور الخاصة بك هي: {generate_password(length)}")

لعبة توقع الرقم

في هذه اللعبة، يقوم البرنامج باختيار رقم عشوائي بين 1 و 100، وعلى اللاعب أن يخمن الرقم. يقدم البرنامج تلميحات حول ما إذا كان الرقم المدخل أعلى أو أقل من الرقم الصحيح. هذا المشروع سيساعدك على تعلم التعامل مع الحلقات والشروط في بايثون.


import random

def guess_number():
    number_to_guess = random.randint(1, 100)
    guess = None
    attempts = 0

    print("توقع الرقم بين 1 و 100")

    while guess != number_to_guess:
        guess = int(input("أدخل تخمينك: "))
        attempts += 1
        if guess < number_to_guess:
            print("أعلى!")
        elif guess > number_to_guess:
            print("أقل!")
        else:
            print(f"مبروك! لقد خمّنت الرقم الصحيح في {attempts} محاولات.")

guess_number()
